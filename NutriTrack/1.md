    %% 实体（表）定义
    users {
        int id PK
        varchar(64) username UK
        varchar(256) password_hash
        varchar(120) email UK
        float height
        float weight
        int age
        varchar(10) gender
        varchar(255) health_goal
        int is_admin
        datetime register_time
        datetime last_login_time
        int status
        float bmi
        varchar(20) bmi_category
        int bmr
        varchar(100) dietary_preference
        varchar(255) allergies
        varchar(255) favorite_foods
        varchar(100) meal_frequency
        varchar(50) exercise_frequency
        varchar(255) exercise_types
        int exercise_duration
        varchar(20) exercise_level
        float target_weight
        datetime goal_deadline
        varchar(20) weight_loss_speed
        int daily_calorie_goal
        text goal_description
    }

    canteens {
        varchar(7) canteen_id PK
        varchar(100) name
    }

    dishes {
        varchar(7) dish_id PK
        varchar(100) name
        varchar(20) cooking_method
        varchar(200) description
        varchar(7) canteen_id FK
    }

    ingredients {
        varchar(7) ingredient_id PK
        varchar(100) ingredient_name
    }

    dish_ingredients {
        varchar(7) dish_id FK
        varchar(7) ingredient_id FK
        float amount_g
        PK(dish_id, ingredient_id)
    }

    nutrition_facts {
        varchar(7) ingredient_id PK,FK
        float energy_kcal
        float protein_g
        float fat_g
        float carb_g
    }

    plate {
        varchar(50) plate_id PK
        int user_id FK
        datetime bind_time
        float current_weight
        int bind_status
    }

    detection_records {
        int id PK
        int user_id FK
        varchar(50) plate_id FK
        datetime bind_time
        float current_weight
        text weight_log
        text detected_objects
        datetime detect_time
    }

    diet_habits {
        int id PK
        int user_id FK
        text habit_content
        datetime create_time
    }

    diet_records {
        int id PK
        int user_id FK
        int meal_type
        text dish_list
        float total_calorie
        float total_protein
        float total_fat
        float total_carb
        datetime create_time
    }

    %% 关系定义（1:N 表示一对多，N:M 表示多对多）
    users ||--o{ plate : "拥有"
    users ||--o{ detection_records : "产生"
    users ||--o{ diet_habits : "拥有"
    users ||--o{ diet_records : "拥有"
    canteens ||--o{ dishes : "提供"
    dishes ||--o{ dish_ingredients : "包含"
    ingredients ||--o{ dish_ingredients : "被包含于"
    ingredients ||--|| nutrition_facts : "对应"
    plate ||--o{ detection_records : "关联"
